 19def nuc_mat_symm_cupy(basis, mol, slice=None, cp_stream=None, sqrt_ints4c2e_diag=None):
 20    # Here the lists are converted to numpy arrays for better use with Numba.
 21    # Once these conversions are done we pass these to a Numba decorated
 22    # function that uses prange, etc. to calculate the matrix efficiently.
 23
 24    # This function calculates the kinetic energy matrix for a given basis object.
 25    # The basis object holds the information of basis functions like: exponents, coeffs, etc.
 26    # It is possible to only calculate a slice (block) of the complete matrix.
 27    # slice is a 4 element list whose first and second elements give the range of the rows to be calculated.
 28    # the third and fourth element give the range of columns to be calculated.
 29    # slice = [start_row, end_row, start_col, end_col]
 30    # The integrals are performed using the formulas
 31
 32    #We convert the required properties to numpy arrays as this is what Numba likes.
 33    bfs_coords = cp.array([basis.bfs_coords])
 34    bfs_contr_prim_norms = cp.array([basis.bfs_contr_prim_norms])
 35    bfs_lmn = cp.array([basis.bfs_lmn])
 36    bfs_nprim = cp.array([basis.bfs_nprim])
 37    coordsBohrs = cp.array([mol.coordsBohrs])
 38    Z = cp.array([mol.Zcharges])
 39    natoms = mol.natoms
 40
 41    #The remaining properties like bfs_coeffs are a list of lists of unequal sizes.
 42    #Numba won't be able to work with these efficiently.
 43    #So, we convert them to a numpy 2d array by applying a trick,
 44    #that the second dimension is that of the largest list. So that
 45    #it can accomadate all the lists.
 46    maxnprim = max(basis.bfs_nprim)
 47    bfs_coeffs = cp.zeros([basis.bfs_nao, maxnprim])
 48    bfs_expnts = cp.zeros([basis.bfs_nao, maxnprim])
 49    bfs_prim_norms = cp.zeros([basis.bfs_nao, maxnprim])
 50    for i in range(basis.bfs_nao):
 51        for j in range(basis.bfs_nprim[i]):
 52            bfs_coeffs[i,j] = basis.bfs_coeffs[i][j]
 53            bfs_expnts[i,j] = basis.bfs_expnts[i][j]
 54            bfs_prim_norms[i,j] = basis.bfs_prim_norms[i][j]
 55        
 56
 57
 58    if slice is None:
 59        slice = [0, basis.bfs_nao, 0, basis.bfs_nao]
 60        
 61    #Limits for the calculation of overlap integrals
 62    a = int(slice[0]) #row start index
 63    b = int(slice[1]) #row end index
 64    c = int(slice[2]) #column start index
 65    d = int(slice[3]) #column end index
 66
 67    # Infer the matrix shape from the start and end indices
 68    num_rows = b - a 
 69    num_cols = d - c 
 70    start_row = a
 71    end_row = b
 72    start_col = c
 73    end_col = d
 74    matrix_shape = (num_rows, num_cols)
 75
 76    # Check if the slice of the matrix requested falls in the lower/upper triangle or in both the triangles
 77    upper_tri = False
 78    lower_tri = False
 79    both_tri_symm = False
 80    both_tri_nonsymm = False
 81    if end_row <= start_col:
 82        upper_tri = True
 83    elif start_row >= end_col:
 84        lower_tri = True
 85    elif start_row==start_col and end_row==end_col:
 86        both_tri_symm = True
 87    else:
 88        both_tri_nonsymm = True
 89
 90    # Initialize the matrix with zeros
 91    V = cp.zeros(matrix_shape) 
 92
 93    if sqrt_ints4c2e_diag is None:
 94        #Create dummy array
 95        sqrt_ints4c2e_diag = cp.zeros((1,1), dtype=cp.float64)
 96        isSchwarz = False
 97    else:
 98        isSchwarz = True
 99        sqrt_ints4c2e_diag = cp.asarray(sqrt_ints4c2e_diag)
100
101    
102
103    if cp_stream is None:
104        device = 0
105        cp.cuda.Device(device).use()
106        cp_stream = cp.cuda.Stream(non_blocking = True)
107        nb_stream = cuda.external_stream(cp_stream.ptr)
108        cp_stream.use()
109    else:
110        nb_stream = cuda.external_stream(cp_stream.ptr)
111        cp_stream.use()
112
113    thread_x = 32
114    thread_y = 32
115    blocks_per_grid = ((num_rows + (thread_x - 1))//thread_x, (num_cols + (thread_y - 1))//thread_y) 
116    nuc_mat_symm_internal_cuda[blocks_per_grid, (thread_x, thread_y), nb_stream](bfs_coords[0], bfs_contr_prim_norms[0], bfs_lmn[0], bfs_nprim[0], bfs_coeffs, bfs_prim_norms, bfs_expnts, a, b, c, d, Z[0], coordsBohrs[0], natoms, lower_tri, upper_tri, both_tri_symm, both_tri_nonsymm, sqrt_ints4c2e_diag, isSchwarz, V)
117    if both_tri_symm:
118        thread_x = 32
119        thread_y = 32
120        blocks_per_grid = ((num_rows + (thread_x - 1))//thread_x, (num_cols + (thread_y - 1))//thread_y) 
121        symmetrize[blocks_per_grid, (thread_x, thread_y), nb_stream](a,b,c,d,V)
122    if cp_stream is None:
123        cuda.synchronize()
124    else:
125        cp_stream.synchronize()
126        cp.cuda.Stream.null.synchronize()
127        # cp._default_memory_pool.free_all_blocks()
128    return V

138def fastFactorial(n):
139    # This is the way to access global constant arrays (which need to be on host, i.e. created using numpy for some reason)
140    # See https://stackoverflow.com/questions/63311574/in-numba-how-to-copy-an-array-into-constant-memory-when-targeting-cuda
141    LOOKUP_TABLE_ = cuda.const.array_like(LOOKUP_TABLE) 
142    # 2-3x faster than the fastFactorial_old for values less than 21
143    if n<= 1:
144        return 1
145    elif n<=20:
146        return LOOKUP_TABLE_[n]
147    else:
148        factorial = 1
149        for i in range(2, n+1):
150            factorial *= i
151        return factorial

160@cuda.jit(fastmath=True, cache=True, device=True)
161def comb(x, y):
162    table = cuda.const.array_like(LOOKUP_TABLE_COMB) 
163    if y == 0: 
164        return 1
165    if x == y: 
166        return 1
167    if x<=5 and y<=5:
168        return table[x,y]
169    binom = fastFactorial(x) // fastFactorial(y) // fastFactorial(x - y)
170    return binom

174@cuda.jit(fastmath=True, cache=True, device=True)   
175def c2k(k,la,lb,PA,PB):
176    temp = 0.0
177    for i in range(la+1):
178        if i>k:
179            continue
180        factor1 = comb(la,i)
181        factor2 = PA**(la-i)
182        for j in range(lb+1):
183            # if j>k:
184            #     continue
185            if (i+j)==k :
186                temp +=  factor1*comb(lb,j)*factor2*PB**(lb-j)
187    return temp
