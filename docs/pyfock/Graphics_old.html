 96def cylinder_between(x1, y1, z1, x2, y2, z2, height, rad):
 97    v = [x2-x1, y2-y1, z2-z1]
 98    axis = (1, 0, 0) if np.hypot(v[0], v[1]) < 0.001 else np.cross(v, (0, 0, 1))
 99    angle = -np.arctan2(np.hypot(v[0], v[1]), v[2])*180/np.pi
100
101    glPushMatrix()
102    glTranslate(x1, y1, z1)
103    glRotate(angle, *axis)
104    quadratic = gluNewQuadric()
105    gluCylinder(quadratic, rad, rad, height, 20, 20)
106    glPopMatrix()

108def calculateBondPositions():
109    global xCoord, yCoord, zCoord, atomicNumber, bondStart, bondEnd, bondLengths
110    bondStart = []
111    bondEnd = []
112    bondLengths = []
113    for i in range(totalNumAtoms):
114        for j in range(i,totalNumAtoms):
115            bondLength = calculateBondLength([xCoord[i],yCoord[i],zCoord[i]],[xCoord[j],yCoord[j],zCoord[j]])
116            if bondLength<=(float(Data.covalentRadius[atomicNumber[i]])+float(Data.covalentRadius[atomicNumber[j]])):
117                bondStart.append(i)
118                bondEnd.append(j)
119                bondLengths.append(bondLength)

122def calculateBondLength(posA, posB):
123    return np.sqrt(np.power(posA[0]-posB[0],2)+np.power(posA[1]-posB[1],2)+np.power(posA[2]-posB[2],2))

281def _create_rotation_matrix(angle, x, y, z):
282    """ Creates a 3x3 rotation matrix. """
283    if la.norm((x, y, z)) < 0.0001:
284        return np.eye(3, dtype=np.float32)
285    x, y, z = np.array((x, y, z))/la.norm((x, y, z))
286    matrix = np.zeros((3, 3), dtype=np.float32)
287    cos = np.cos(angle)
288    sin = np.sin(angle)
289    matrix[0, 0] = x*x*(1-cos)+cos
290    matrix[1, 0] = x*y*(1-cos)+sin*z
291    matrix[0, 1] = x*y*(1-cos)-sin*z
292    matrix[2, 0] = x*z*(1-cos)-sin*y
293    matrix[0, 2] = x*z*(1-cos)+sin*y
294    matrix[1, 1] = y*y*(1-cos)+cos
295    matrix[1, 2] = y*z*(1-cos)-sin*x
296    matrix[2, 1] = y*z*(1-cos)+sin*x
297    matrix[2, 2] = z*z*(1-cos)+cos
298    return matrix

339def _mouse_move_callback(x, y):
340    """ Mouse move event handler for GLUT. """
341    global _previous_mouse_position, camera, eye, center, up, maxDistance, _mouse_dragging
342    if _mouse_dragging:
343        width = glutGet(GLUT_WINDOW_WIDTH)
344        height = glutGet(GLUT_WINDOW_HEIGHT)
345        dx = (x-_previous_mouse_position[0])/width
346        dy = (y-_previous_mouse_position[1])/height
347        glRotatef(dx,1,0,0)
348        glRotatef(dy,0,1,0)
349        rotation_intensity = la.norm((dx, dy)) * 4
350        eye, center, up = camera
351        camera_distance = la.norm(center-eye)
352        forward = (center-eye)/camera_distance
353        right = np.cross(forward, up)
354        rotation_axis = (up*dx+right*dy)
355        rotation_matrix = _create_rotation_matrix(-rotation_intensity,
356                                                  rotation_axis[0],
357                                                  rotation_axis[1],
358                                                  rotation_axis[2])
359        forward = np.dot(rotation_matrix, forward)
360        up = np.dot(rotation_matrix, up)
361        eye = center-forward*camera_distance
362        camera = eye, center, up
363        _previous_mouse_position = (x, y)
364        #print(_mouse_dragging)
365        # Set matrix mode
366        glMatrixMode(GL_MODELVIEW)
367        # Reset matrix
368        glLoadIdentity()
369        #eye[2] = -maxDistance
370        gluLookAt(eye[0], eye[1], eye[2],
371                         center[0], center[1], center[2],
372                         up[0], up[1], up[2])
373        glutPostRedisplay()
